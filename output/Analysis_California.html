Background and Objectives

This markdown file uses ggplot2 (Wickham 2016) and dplyr(Wickham et al. 2020) packages to analyze the Apple Maps mobility data from the time before the COVID-19 pandemic through the initial shutdowns and continuing changes over the course of the year.

The goal is to create a subset to a particular state, tally up the number of cities and counties in that state, and then to make several plots to visualize that data.

The state analyzed in this document is California.

Data Processing

# subset out to the chosen state using the variables set above
state_subset <- subset_mobility_data_to_state(
  input_filename = global_file_to_process,
  subregion = state_to_analyze)
# convert the subsetted state data from wide to long format
long_subset <- flip_dataset(
  file_name_in = paste0("output/subsetted_states_wide/",

# tally up the numbers of cities and counties in the state that have mobility

tallied_subset <- count_cities_counties(
  file_name_in = paste0("output/subsetted_states_wide/",
                      state_no_spaces,

# produce a line plot of relative driving mobility across the state with data
# on the x axis and mean relative mobility on the y axis
timeseries_plot <- long_subset %>%
  filter(transportation_type == "driving") %>%
  group_by(date) %>%
  summarize(mean_mobility = mean(rel_mobility)) %>%
  ggplot(aes(x = lubridate::ymd(date),
             y = mean_mobility)) +
  geom_line() +
  labs(title = paste("Statewide mean relative mobility driving levels in",
                     state_to_analyze,
                     "during COVID"),
       x = "Date",
       y = "Mean relative mobility")

ggsave(plot = timeseries_plot,
       filename = paste0("output/figures/",
                         state_no_spaces,
                         "_",
                         core_name,
                         "_timeseries_plot.png"))

timeseries_plot

# read in the count of the sequences by country (2 columns)
seqs_by_country <- readr::read_table(params$seqdata,
                                     col_names = c("count",
                                                   "country"))
# check for two columns
stopifnot(ncol(seqs_by_country) == 2)

# make a sideways barplot, log scale x axis, with countries on the y
# and the counts of sequences in each on the x, sorted by count
seq_count_plot <- ggplot(data = seqs_by_country,
                         aes(x = count,
                             y = reorder(country, count))) +
  geom_col() +
  scale_x_log10() +
  labs(title = "Count of SARS-CoV-2 sequences per country",
       x = "Count of Sequences",
       y = "Country Name")

ggsave(plot = seq_count_plot,
       filename = "output/figures/seq_count_plot.png")

seq_count_plot

# produce pre- and post- lockdown graphs of 3 lines each, one line for each
# transit type, with day of the week on the x axis and mean relative mobility
# on the y axis. The first graph should be from January and the second graph
# should be aggregate from March to April
lockdown_plot <- long_subset %>%
  group_by(date) %>%
  summarize(mean_mobility = mean(rel_mobility)) %>%
  ggplot(aes(x = lubridate::wday(date, label = TRUE),
             y = mean_mobility)) +
  geom_boxplot() +
  labs(title = paste("Weekly statewide mean relative mobility levels in",
                     state_to_analyze,
                     "before and during COVID"),
       x = paste("Day of the Week"),
       y = "Mean relative mobility") +
  facet_wrap(. ~ factor(ifelse(lubridate::month(date) < 3,
                        "Pre-COVID", # facet label for TRUE
                        "During COVID"), # facet label for FALSE
                        levels = c("Pre-COVID", # switch order so pre goes first
                                   "During COVID")))

ggsave(plot = lockdown_plot,
       filename = paste0("output/figures/",
                         state_no_spaces,
                         "_",
                         core_name,
                         "_lockdown_plot.png"))
